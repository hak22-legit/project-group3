cmake_minimum_required(VERSION 3.20)
cmake_policy(SET CMP0135 NEW)
project(MediaCatalog VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

FetchContent_Declare(
    nlohmann_json
    URL https://github.com/nlohmann/json/releases/download/v3.11.3/json.tar.xz
    URL_HASH SHA256=d6c65aca6b1ed68e7a182f4757257b107ae403032760ed6ef121c9d55e81757d
)
FetchContent_MakeAvailable(nlohmann_json)

FetchContent_Declare(
    tabulate
    GIT_REPOSITORY https://github.com/p-ranav/tabulate.git
    GIT_TAG        master
)

FetchContent_MakeAvailable(tabulate)


option(ENABLE_HTTPLIB "Enable OMDb fetch" OFF)
message(STATUS "ENABLE_HTTPLIB: ${ENABLE_HTTPLIB}")

add_executable(media_catalog
    src/main.cpp
    src/auth.cpp
    src/catalog.cpp
    src/input.cpp
    src/storage.cpp
    src/omdb.cpp
)

target_include_directories(media_catalog PRIVATE
    include
    vendor
    ${tabulate_SOURCE_DIR}/single_include
)

target_link_libraries(media_catalog PRIVATE
    nlohmann_json::nlohmann_json
    tabulate::tabulate
)

target_compile_definitions(media_catalog PRIVATE
    WIN32_LEAN_AND_MEAN
    NOMINMAX
)

if(ENABLE_HTTPLIB)
    target_compile_definitions(media_catalog PRIVATE
        HTTPLIB_AVAILABLE
    )
    target_link_libraries(media_catalog PRIVATE
        ws2_32
    )
endif()

set_target_properties(media_catalog PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}"
)
